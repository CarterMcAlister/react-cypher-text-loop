## Install

```bash
npm install --save react-cypher-text
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-cypher-text'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
